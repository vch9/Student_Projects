let rec print_value fmt v = match v with | VInt n -> Format.fprintf fmt "%d" n | VBool b -> Format.fprintf fmt "%b" b | VPair (v, w) -> Format.fprintf fmt "@[(%a,@ %a)@]" print_value v print_value w | VClo (x, m, env) -> Format.fprintf fmt "@[clo(%a,@ %a,@ %a)@]" print_id x print_term m print_environment env
(** print_environment fmt env affiche l'environnement env en utilisant le formateur fmt. *)
and print_environment fmt env = print_id_map print_value fmt env
